package main

import (
  "fmt"
  "io"
  "log"
  "net/http"
  "strings"
)

func main() {
  client := &http.Client{}
  var data = strings.NewReader(`{
        "input" : {
          "description":"Her name is Geeta. She looks cute and has a Heart of Gold (metaphor)!",
          "response_type":"Funny, Flirty"}
     }`)
  req, err := http.NewRequest("POST", "http://localhost:8000/pickup/invoke", data)
  if err != nil {
    log.Fatal(err)
  }
  req.Header.Set("Content-Type", "application/json")
  resp, err := client.Do(req)
  if err != nil {
    log.Fatal(err)
  }
  defer resp.Body.Close()
  bodyText, err := io.ReadAll(resp.Body)
  if err != nil {
    log.Fatal(err)
  }
  fmt.Printf("%s\n", bodyText)
}
